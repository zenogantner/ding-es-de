mis :: meine
tu :: dein; deine
tus :: deine
su :: sein; seine; ihr; ihre
sus :: seine; ihre
nuestro; nuestra :: unser; unsere
vuestro; vuestra :: euer; eure

mío; mía :: mein; meine; meiner
tuyo; tuya :: dein; deine; deiner
suyo; suya :: sein; seine; seiner
suyo; suya :: ihr; ihre; ihrer
suyo; suya :: Ihr; Ihre; Ihrer
nuestros; nuestras :: unsere
vuestros; vuestras :: eure

la nuestra :: die Unsere
las nuestras :: die Unseren
el nuestro :: der Unsere
los nuestros :: die Unseren
lo nuestro :: das Unsere
la vuestra :: die Eure
el vuestro :: der Eure
lo vuestro :: das Eure
